async function processCards(cards){ const zip = new JSZip(); let index = []; let done = 0; async function worker(queue){ while (queue.length){ const card = queue.shift(); try { await handle(card); } catch (e) { console.error("Card failed:", card?.name, e); } done++; updateBar(done / cards.length); } } async function handle(card){ const url = getImageUrl(card); if (!url) return; const res = await fetch(url); if (!res.ok) return; const blob = await res.blob(); // safer image decoding (fallback chain) let img; try { img = await createImageBitmap(blob); } catch (e) { // fallback for browsers that fail bitmap decoding const bmp = await new Promise((resolve, reject) => { const imgEl = new Image(); imgEl.onload = () => resolve(imgEl); imgEl.onerror = reject; imgEl.src = URL.createObjectURL(blob); }); img = bmp; } const canvas = document.createElement('canvas'); canvas.width = 320; canvas.height = 480; const ctx = canvas.getContext('2d'); ctx.drawImage(img, 0, 0, 320, 480); const base = buildBaseName(card); nameCounts[base] = (nameCounts[base] || 0) + 1; const filename = `${base}_${nameCounts[base]}.png`; const fileBlob = await new Promise(resolve => { canvas.toBlob(resolve, "image/png"); }); if (!fileBlob) { console.warn("Empty blob for", card.name); return; } zip.file(filename, fileBlob); const parts = filename.replace('.png','').split('_'); const version = parts.pop(); const color = parts.pop(); let power='token', tough='token', name; if (parts[parts.length-1] === 'token'){ name = parts.slice(0,-1).join('_'); } else { tough = parts.pop(); power = parts.pop(); name = parts.join('_'); } index.push(`${filename},${name},${power},${tough},${color},${version}`); } const queue = [...cards]; const workers = []; for (let i = 0; i < CONCURRENT; i++){ workers.push(worker(queue)); } await Promise.all(workers); zip.file('index.csv', index.join('\n')); const content = await zip.generateAsync({ type:'blob' }); saveAs(content, 'tokens.zip'); }
